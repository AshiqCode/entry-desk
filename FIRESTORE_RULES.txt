// Firestore Security Rules for ClassTrack
// Copy and paste these rules into Firebase Console > Firestore Database > Rules

// Option 1: Allow any authenticated user to write (recommended for starting)
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /classes/{docId} {
      // Anyone can read classes
      allow read: if true;
      
      // Any authenticated user can create, update, or delete
      allow write: if request.auth != null;
    }
  }
}

// Option 2: Restrict to specific admin UID (more secure)
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /classes/{docId} {
      allow read: if true;
      
      // Replace YOUR_ADMIN_UID with actual admin user's UID from Firebase Console
      allow write: if request.auth != null && request.auth.uid == "YOUR_ADMIN_UID";
    }
  }
}

// Option 3: Use custom claims (most flexible, requires Firebase Functions)
// First, set custom claim using Firebase Admin SDK:
// admin.auth().setCustomUserClaims(uid, { admin: true });

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /classes/{docId} {
      allow read: if true;
      
      // Only users with admin custom claim can write
      allow write: if request.auth != null && request.auth.token.admin == true;
    }
  }
}
